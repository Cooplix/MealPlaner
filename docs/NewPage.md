const EXPIRY_SOON_DAYS = 14;
function computeStatus(item: InventoryItem) {
  let expiryStatus = null;
  if (item.expiresAt) {
    const diffMs = new Date(item.expiresAt).getTime() - Date.now();
    const days = Math.ceil(diffMs / (1000*60*60*24));
    if (days < 0) expiryStatus = "EXPIRED";
    else if (days <= 14) expiryStatus = "<=14d";
    else if (days <= 30) expiryStatus = "<=30d";
    else expiryStatus = "OK";
  }
  let restockStatus = null;
  if (item.minQty !== undefined && item.maxQty !== undefined) {
    restockStatus = item.quantity < item.minQty ? "NEED_RESTOCK" : "OK";
  }
  return { expiryStatus, restockStatus };
}
